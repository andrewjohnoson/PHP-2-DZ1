CREATE TABLE news
(
    id BIGINT auto_increment,
    constraint news_pk
        primary key (id),
    title VARCHAR(100) NOT NULL,
    content TEXT
);

CREATE TABLE users
(
    id BIGINT auto_increment,
    constraint users_pk
        primary key (id),
    email VARCHAR(255) NOT NULL,
    name VARCHAR(100)
);
